===script===
      $(function(){
        var salutation = $('#salutation');
        var name = $('#name');
        var greeting = $('#greeting');

        function update(){
          greeting.text(
              salutation.val() + ' ' +
              name.val() + '!');
        }

        update();

        name.bind('keydown', function(){
          setTimeout(update, 0);
        });
        salutation.bind('change', update);
      });
      ===/script===
