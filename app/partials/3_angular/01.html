      function Greeter($xhr){
        var self = this;

        this.name = 'World';

        this.greet = function() {
          var url = 'http://angularjs.org/greet.php';
          url += '?callback=JSON_CALLBACK';
          url += '&name=' + this.name;
          $xhr('JSON', url, function(code, response){
            self.greeting = response.greeting;
          });
        }
      }
