      angular.widget('my:time', function(element){
        element.css('display', 'block');
        var format = element.text();
        var dateFilter = angular.filter('date');
        return function($defer, element){
          (function tick(){
            element.text(dateFilter(new Date(), format));
            $defer(tick, 1000);
          })();
        };
      });
